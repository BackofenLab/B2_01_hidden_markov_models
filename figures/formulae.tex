&\iff \frac{1}{1 + 49 * (\frac{1}{3})^n} &&\geq \frac{9}{10} &&\quad| \text{ simplify, given } n > 0\\
&\iff \frac{9}{10} (1 + 49 * (\frac{1}{3})^n) &&\leq 1  &&\quad| \text{ cross-multiply, given } n > 0\\
&\iff (\frac{1}{3})^n &&\leq \frac{1}{441} &&\quad| \text{ rewrite }\\
